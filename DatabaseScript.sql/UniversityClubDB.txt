
CREATE DATABASE UniversityClubDB;
GO

USE UniversityClubDB;
GO

CREATE TABLE Students
(
    StudentId INT IDENTITY(1,1) PRIMARY KEY,
    FullName NVARCHAR(100) NOT NULL,
    Email NVARCHAR(100) NOT NULL UNIQUE,
    Username NVARCHAR(50) NOT NULL UNIQUE,
    PasswordHash NVARCHAR(255) NOT NULL,
    Phone NVARCHAR(20) NULL,
    Major NVARCHAR(100) NULL,
    IsBlocked BIT NOT NULL DEFAULT 0,
    CreatedDate DATETIME NOT NULL DEFAULT GETDATE()
);
GO

CREATE TABLE Admins
(
    AdminId INT IDENTITY(1,1) PRIMARY KEY,
    Username NVARCHAR(50) NOT NULL UNIQUE,
    PasswordHash NVARCHAR(255) NOT NULL
);
GO

CREATE TABLE Clubs
(
    ClubId INT IDENTITY(1,1) PRIMARY KEY,
    ClubName NVARCHAR(100) NOT NULL,
    Description NVARCHAR(MAX) NULL,
    Category NVARCHAR(100) NOT NULL,
    PresidentName NVARCHAR(100) NOT NULL,
    CreatedDate DATETIME NOT NULL DEFAULT GETDATE(),
    IsActive BIT NOT NULL DEFAULT 1
);
GO

CREATE TABLE Activities
(
    ActivityId INT IDENTITY(1,1) PRIMARY KEY,
    ClubId INT NOT NULL,
    ActivityTitle NVARCHAR(100) NOT NULL,
    Description NVARCHAR(MAX) NULL,
    ActivityDate DATETIME NOT NULL,
    Location NVARCHAR(100) NOT NULL,
    CreatedDate DATETIME NOT NULL DEFAULT GETDATE(),
    CONSTRAINT FK_Activities_Clubs FOREIGN KEY (ClubId) REFERENCES Clubs(ClubId) ON DELETE CASCADE
);
GO

CREATE TABLE Memberships
(
    MembershipId INT IDENTITY(1,1) PRIMARY KEY,
    StudentId INT NOT NULL,
    ClubId INT NOT NULL,
    JoinDate DATETIME NOT NULL DEFAULT GETDATE(),
    Status NVARCHAR(50) NOT NULL DEFAULT 'Pending',
    CONSTRAINT FK_Memberships_Students FOREIGN KEY (StudentId) REFERENCES Students(StudentId) ON DELETE CASCADE,
    CONSTRAINT FK_Memberships_Clubs FOREIGN KEY (ClubId) REFERENCES Clubs(ClubId) ON DELETE NO ACTION
);
GO
